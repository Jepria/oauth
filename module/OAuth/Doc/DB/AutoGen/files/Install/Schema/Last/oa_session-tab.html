oa_session

create table oa_session (
   session_id integer not null ,
   auth_code varchar2(32) not null ,
   client_id integer ,
   redirect_uri varchar2(4000) ,
   operator_id integer ,
   code_challenge varchar2(128) ,
   access_token varchar2(32) ,
   access_token_date_ins timestamp with time zone ,
   access_token_date_finish timestamp with time zone ,
   refresh_token varchar2(32) ,
   refresh_token_date_ins timestamp with time zone ,
   refresh_token_date_finish timestamp with time zone ,
   session_token varchar2(32) ,
   session_token_date_ins timestamp with time zone ,
   session_token_date_finish timestamp with time zone ,
   is_manual_blocked number(1) ,
   date_finish timestamp with time zone ,
   date_ins timestamp with time zone default systimestamp not null ,
   operator_id_ins integer not null ,
   constraint oa_session_pk primary key ( session_id) using index tablespace &indexTablespace ,
   constraint oa_session_uk_auth_code unique ( auth_code) using index tablespace &indexTablespace ,
   constraint oa_session_uk_access_token unique ( access_token) using index tablespace &indexTablespace ,
   constraint oa_session_uk_refresh_token unique ( refresh_token) using index tablespace &indexTablespace ,
   constraint oa_session_ck_is_manual_bl_dt check (is_manual_blocked = 1)
)
